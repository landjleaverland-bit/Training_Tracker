60export function getClimbingVsRestStats(sessions: Session[], dateRange?: { start: Date, end: Date }): ChartDataPoint[] {
61 // Filter for valid dates first
62 const validSessions = sessions.filter(s => isValidDate(s.date));
63 if (validSessions.length === 0) return [];
64
65 let firstDate: Date;
66 let lastDate: Date;
67
68 if (dateRange) {
69 firstDate = new Date(dateRange.start);
70 lastDate = new Date(dateRange.end);
71 } else {
72 const sorted = [...validSessions].sort((a, b) => new Date(a.date).getTime() - new Date(b.date).getTime());
73 firstDate = new Date(sorted[0].date);
74 lastDate = new Date(); // Today
75 }
76
77 // Calculate total days in range
78 const msPerDay = 1000 * 60 * 60 * 24;
79
80 // Ensure we count inclusive of both start and end, and handle time of day
81 firstDate.setHours(0, 0, 0, 0);
82 lastDate.setHours(23, 59, 59, 999);
83
84 const totalDays = Math.ceil((lastDate.getTime() - firstDate.getTime()) / msPerDay);
85
86 // Count climbing days (unique dates where activity is climb)
87 // Filter sessions to ensure they are within range (though sessions passed in should ideally be filtered already,
88 // strictly enforcing range helps if mixed data is passed)
89 const climbingDays = new Set(
90 sessions
91 .filter(s => {
92 if (!isClimbing(s)) return false;
93 if (!dateRange) return true;
94 const d = new Date(s.date);
95 return d >= firstDate && d <= lastDate;
96 })
97 .map(s => s.date.split('T')[0])
98 ).size;
99
100 const restDays = Math.max(0, totalDays - climbingDays);
101
102 return [
103 { label: 'Climbing Days', value: climbingDays, color: '#4A9B9B' },
104 { label: 'Rest Days', value: restDays, color: '#E5E7EB' }
105 ];
106}

255export function getGradeStats(sessions: Session[], type: 'boulder' | 'lead', location: 'indoor' | 'outdoor' | 'all' = 'all'): ChartDataPoint[] {
256 // Structure: Grade -> { total: number, flash: number, redpoint: number, dogged: number, dnf: number }
257 const grades: Record<string, { total: number, flash: number, redpoint: number, dogged: number, dnf: number }> = {};
258
259 sessions.forEach(s => {
260 if (!isClimbing(s)) return;
261
262 // Filter by location
263 if (location === 'indoor' && s.activityType !== 'indoor_climb') return;
264 if (location === 'outdoor' && s.activityType !== 'outdoor_climb') return;
265
266 const climbSession = s as IndoorClimbSession | OutdoorClimbSession;
267
268 climbSession.climbs?.forEach(c => {
269 // Check for explicit climbing type, otherwise fallback to isSport
270 let isBoulderItem = !c.isSport;
271 let isLeadItem = c.isSport;
272
273 // Cast to any to check for climbingType that might exist in data but not type definition
274 const typeTag = (c as any).climbingType;
275 if (typeTag) {
276 if (typeTag === 'Boulder') {
277 isBoulderItem = true;
278 isLeadItem = false;
279 } else if (['Sport', 'Trad', 'Lead'].includes(typeTag)) {
280 isBoulderItem = false;
281 isLeadItem = true;
282 }
283 }
284
285 // Normalize grade case (v2 -> V2)
286 let gradeKey = c.grade.toUpperCase();
287
288 // Apply conversions if it's a boulder grade
289 if (type === 'boulder' && isBoulderItem) {
290 // If it's a known French grade, convert it
291 if (FRENCH_TO_HUECO[gradeKey]) {
292 gradeKey = FRENCH_TO_HUECO[gradeKey];
293 }
294 }
295
296 // Apply conversions if it's a lead grade (Sport)
297 if (type === 'lead' && isLeadItem) {
298 // If it's a known Hueco grade, convert it to French
299 if (HUECO_TO_FRENCH[gradeKey]) {
300 gradeKey = HUECO_TO_FRENCH[gradeKey];
301 }
302 }
303
304 // Determine specific type
305 if ((type === 'lead' && isLeadItem) || (type === 'boulder' && isBoulderItem)) {
306 if (!grades[gradeKey]) {
307 grades[gradeKey] = { total: 0, flash: 0, redpoint: 0, dogged: 0, dnf: 0 };
308 }
309
310 grades[gradeKey].total++;
311
312 const attempt = c.attemptType ? c.attemptType.trim().toLowerCase() : '';
313
314 if (['flash', 'onsight'].includes(attempt)) {
315 grades[gradeKey].flash++;
316 } else if (attempt === 'redpoint') {
317 grades[gradeKey].redpoint++;
318 } else if (attempt === 'dogged') {
319 grades[gradeKey].dogged++;
320 } else if (attempt === 'dnf') {
321 grades[gradeKey].dnf++;
322 } else {
323 // Fallback for unknown types (count as redpoint? or just ignore breakdown but keep usage in total?)
324 // Let's count as redpoint for backward compatibility if "attemptType" is missing or odd
325 grades[gradeKey].redpoint++;
326 }
327 }
328 });
329 });
330
331 return Object.entries(grades)
332 .map(([label, stats]) => {
333 // Convert label to lowercase for French grades (e.g. 7A+ -> 7a+)
334 const displayLabel = /^\d+[A-C][+]?$/.test(label) ? label.toLowerCase() : label;
335 return {
336 label: displayLabel,
337 value: stats.total,
338 flashCount: stats.flash,
339 redpointCount: stats.redpoint,
340 doggedCount: stats.dogged,
341 dnfCount: stats.dnf
342 };
343 })
344 .sort((a, b) => a.label.localeCompare(b.label, undefined, { numeric: true }));
345}

402export function getWeeklyLoadStats(sessions: Session[]): TimeSeriesPoint[] {
403 // Group by week
404 const weeks: Record<string, number> = {};
405 const weekFormat = d3.timeFormat("%Y-%W");
406
407 sessions.forEach(s => {
408 const date = getSessionDate(s);
409 const weekKey = weekFormat(date); // e.g., "2024-05"
410
411 // Calculate abstract load (sum of finger/shoulder/forearm load)
412 let load = 0;
413 if (isClimbing(s)) {
414 const c = s as IndoorClimbSession; // or Outdoor, fields satisfy structural shape
415 load = (c.fingerLoad || 0) + (c.shoulderLoad || 0) + (c.forearmLoad || 0);
416 } else if (s.activityType === 'competition') {
417 const c = s as unknown as any; // CompetitionSession
418 load = (c.fingerLoad || 0) + (c.shoulderLoad || 0) + (c.forearmLoad || 0);
419 }
420
421 weeks[weekKey] = (weeks[weekKey] || 0) + load;
422 });
423
424 // Convert back to array
425 const weekParse = d3.timeParse("%Y-%W");
426 return Object.entries(weeks).map(([key, value]) => ({
427 date: weekParse(key) || new Date(),
428 value,
429 series: 'Total Load'
430 })).sort((a, b) => a.date.getTime() - b.date.getTime());
431}

557export function movingAverage(data: TimeSeriesPoint[], windowSize: number): TimeSeriesPoint[] {
558 if (data.length === 0) return [];
559
560 // Sort just in case
561 const sorted = [...data].sort((a, b) => a.date.getTime() - b.date.getTime());
562 const averaged: TimeSeriesPoint[] = [];
563
564 for (let i = 0; i < sorted.length; i++) {
565 const start = Math.max(0, i - windowSize + 1);
566 const windowSlice = sorted.slice(start, i + 1);
567 const sum = windowSlice.reduce((acc, curr) => acc + curr.value, 0);
568 const avg = sum / windowSlice.length;
569
570 averaged.push({
571 date: sorted[i].date,
572 value: avg,
573 series: `${sorted[i].series} (Avg)`
574 });
575 }
576
577 return averaged;
578}

591export function getLoadStats(
592 sessions: Session[],
593 filterFn: (s: Session) => boolean,
594 metricFn: (s: Session) => number,
595 seriesName: string,
596 groupBy: 'week' | 'day' = 'week',
597 dateRange?: { start: Date, end: Date }
598): TimeSeriesPoint[] {
599 const groups: Record<string, number> = {};
600 const format = groupBy === 'week' ? d3.timeFormat("%Y-%W") : d3.timeFormat("%Y-%m-%d");
601
602 // 1. Populate with actual data
603 sessions.forEach(s => {
604 // First check if session matches the criteria (e.g. is Indoor Climb)
605 if (!filterFn(s)) return;
606
607 const date = getSessionDate(s);
608 const key = format(date);
609
610 // Then extract the metric (e.g. 1 for count, or s.fingerLoad for specific load)
611 const val = Number(metricFn(s)) || 0;
612
613 groups[key] = (groups[key] || 0) + val;
614 });
615
616 // 2. Zero-fill if range is provided
617 if (dateRange) {
618 let current = new Date(dateRange.start);
619 const end = new Date(dateRange.end);
620
621 // Normalize start to beginning of day to avoid infinite loops with time offsets
622 current.setHours(0, 0, 0, 0);
623 end.setHours(23, 59, 59, 999);
624
625 // Safety break: don't loop more than 366 * 10 iterations
626 let iterations = 0;
627 const maxIterations = 4000;
628
629 while (current <= end && iterations < maxIterations) {
630 const key = format(current);
631 // Only set if not already present (preserve actual data)
632 if (groups[key] === undefined) {
633 groups[key] = 0;
634 }
635 // Increment by 1 day
636 current.setDate(current.getDate() + 1);
637 iterations++;
638 }
639 }
640
641 const parse = groupBy === 'week' ? d3.timeParse("%Y-%W") : d3.timeParse("%Y-%m-%d");
642
643 return Object.entries(groups).map(([key, value]) => ({
644 date: parse(key) || new Date(),
645 value,
646 series: seriesName
647 })).sort((a, b) => a.date.getTime() - b.date.getTime());
