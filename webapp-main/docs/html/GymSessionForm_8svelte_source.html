103 function getBenchmarks(exerciseName: string) {

108 // Sort sessions descending by date (newest first)

111 );

113 for (const session of sortedSessions) {
114 // Optimization: if all filled, break
115 if (result.Green && result.Yellow && result.Orange && result.Red) break;
116
117 const ex = session.exercises.find(e => e.name === exerciseName);
118 if (ex && ex.difficulty && !result[ex.difficulty]) {
119 // Find best set (Max weight)
120 let bestSet: GymSet | null = null;
121 for (const set of ex.sets) {
122 if (!bestSet || set.weight > bestSet.weight) {
123 bestSet = set;

125 bestSet = set;
126 }
127 }
128
129 if (bestSet) {
130 result[ex.difficulty] = { weight: bestSet.weight, reps: bestSet.reps };
131 }
132 }
133 }
