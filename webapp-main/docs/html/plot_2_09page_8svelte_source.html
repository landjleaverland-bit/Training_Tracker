111 const getISOWeek = (d: Date) => {
112 const date = new Date(d.getTime());
113 date.setHours(0, 0, 0, 0);
114 date.setDate(date.getDate() + 3 - (date.getDay() + 6) % 7);
115 const week1 = new Date(date.getFullYear(), 0, 4);
116 return 1 + Math.round(((date.getTime() - week1.getTime()) / 86400000 - 3 + (week1.getDay() + 6) % 7) / 7);
117 };
118
119 return sessions.filter(s => {
120 const d = new Date(s.date);
121 return d.getFullYear() === year && getISOWeek(d) === week;
122 });

153
154 // Calculate start of ISO week
155 const simple = new Date(year, 0, 1 + (week - 1) * 7);
156 const dow = simple.getDay();
157 const ISOweekStart = simple;
158 if (dow <= 4)
159 ISOweekStart.setDate(simple.getDate() - simple.getDay() + 1);
160 else
161 ISOweekStart.setDate(simple.getDate() + 8 - simple.getDay());
162
163 const wStart = new Date(ISOweekStart);
164 const wEnd = new Date(ISOweekStart);
165 wEnd.setDate(wEnd.getDate() + 6);
166 return { start: wStart, end: wEnd };

168 } else if (timeRange === 'specific_month' && selectedDateValue) {
169 const [year, month] = selectedDateValue.split('-');

172 return { start: mStart, end: mEnd };
